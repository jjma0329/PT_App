// Creates a 1-hour Google Calendar event for a confirmed booking.
// Called after the Booking row is saved to the DB.
// Failure is non-fatal — if this throws, the caller catches it and logs it.
export async function createCalendarEvent(details: {
  name: string;
  email: string;
  slotTime: Date;
}): Promise<void> {
  const timezone = process.env.TRAINER_TIMEZONE ?? 'UTC';
  const authClient = await getAuthenticatedClient();
  const calendar = google.calendar({ version: 'v3', auth: authClient });

  const startTime = details.slotTime.toISOString();
  // Sessions are 1 hour — compute end time
  const endTime = new Date(details.slotTime.getTime() + 60 * 60 * 1000).toISOString();

  await calendar.events.insert({
    calendarId: 'primary',
    requestBody: {
      summary: `PT Session — ${details.name}`,
      description: `Booked by: ${details.name}\nEmail: ${details.email}`,
      start: { dateTime: startTime, timeZone: timezone },
      end: { dateTime: endTime, timeZone: timezone },
    },
  });
}
```

Do not remove or modify any existing functions (`createOAuthClient`, `saveTokensFromCode`, `getAuthenticatedClient`, `getAvailableSlots`).

// Shared shape for both booking email functions
interface BookingDetails {
  name: string;
  email: string;
  phone?: string | null;
  message?: string | null;
  slotTime: Date;
}

// Formats slotTime as a readable local time string for email display.
// Uses TRAINER_TIMEZONE so the time shown matches the trainer's local clock.
function formatSlotTime(slotTime: Date): string {
  return slotTime.toLocaleString('en-US', {
    timeZone: process.env.TRAINER_TIMEZONE ?? 'UTC',
    weekday: 'long',
    month: 'long',
    day: 'numeric',
    hour: 'numeric',
    minute: '2-digit',
  });
}

// Sends booking confirmation to the visitor.
// Subject: "Booking Confirmed — {name}"
export async function sendVisitorConfirmation(booking: BookingDetails): Promise<void> {
  const formattedTime = formatSlotTime(booking.slotTime);
  await resend.emails.send({
    from: 'JJM Fitness <onboarding@resend.dev>',
    to: booking.email,
    subject: `Booking Confirmed — ${escapeHtml(booking.name)}`,
    html: `
      <h2>Your session is confirmed!</h2>
      <p>Hi ${escapeHtml(booking.name)},</p>
      <p>Your training session has been booked for:</p>
      <p><strong>${formattedTime}</strong></p>
      <p>If you need to reschedule or have questions, reply to this email.</p>
      <p>— JJM Fitness</p>
    `,
  });
}

// Sends booking notification to the trainer.
// Subject: "New Booking — {name}"
export async function sendTrainerNotification(booking: BookingDetails): Promise<void> {
  const formattedTime = formatSlotTime(booking.slotTime);
  await resend.emails.send({
    from: 'JJM Fitness <onboarding@resend.dev>',
    to: process.env.TRAINER_EMAIL!,
    subject: `New Booking — ${escapeHtml(booking.name)}`,
    html: `
      <h2>New Session Booking</h2>
      <table cellpadding="8" style="border-collapse:collapse;font-family:sans-serif;font-size:15px;">
        <tr><td><strong>Name</strong></td><td>${escapeHtml(booking.name)}</td></tr>
        <tr><td><strong>Email</strong></td><td>${escapeHtml(booking.email)}</td></tr>
        <tr><td><strong>Phone</strong></td><td>${safeField(booking.phone)}</td></tr>
        <tr><td><strong>Time</strong></td><td>${formattedTime}</td></tr>
        <tr><td><strong>Message</strong></td><td>${safeField(booking.message)}</td></tr>
      </table>
    `,
  });
}
```

The `resend` client, `escapeHtml`, and `safeField` are already defined in the file — reference them directly, do not redefine them.
