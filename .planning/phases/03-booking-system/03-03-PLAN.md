**Create `server/controllers/bookingsController.ts`** — follows the write order: validate → re-verify availability → DB insert → Calendar (non-fatal) → emails (non-fatal) → 201.

```typescript
import { Request, Response } from 'express';
import { prisma } from '../lib/prisma.ts';
import { Prisma } from '../../src/generated/prisma/index.js';
import { getAvailableSlots, createCalendarEvent } from '../services/calendarService.ts';
import { sendVisitorConfirmation, sendTrainerNotification } from '../services/emailService.ts';

// POST /api/bookings
// Body: { name, email, phone?, message?, slotTime }
// slotTime must be an ISO 8601 string matching a slot returned by /api/slots
export async function createBooking(req: Request, res: Response): Promise<void> {
  const { name, email, phone, message, slotTime } = req.body as {
    name?: string;
    email?: string;
    phone?: string;
    message?: string;
    slotTime?: string;
  };

  // 1. Input validation — fail fast with a clear message
  if (!name || !email || !slotTime) {
    res.status(400).json({ success: false, error: 'name, email, and slotTime are required.' });
    return;
  }

  // 2. Validate slotTime is a parseable date before doing anything else
  const slotDate = new Date(slotTime);
  if (isNaN(slotDate.getTime())) {
    res.status(400).json({ success: false, error: 'slotTime must be a valid ISO date string.' });
    return;
  }

  // 3. Re-verify the slot is still available on the server side.
  //    The visitor's UI may be stale — another booking could have taken this slot
  //    in the time since the slots page loaded.
  const dateStr = slotTime.slice(0, 10); // "2026-04-01T18:00:00.000Z" → "2026-04-01"
  try {
    const availableSlots = await getAvailableSlots(dateStr);
    if (!availableSlots.includes(slotTime)) {
      res.status(409).json({ success: false, error: 'This time slot is no longer available. Please choose another.' });
      return;
    }
  } catch {
    // If the calendar check fails entirely, proceed to the DB write.
    // The @unique constraint is the authoritative guard.
  }

  try {
    // 4. Write the booking to the DB first — this is the authoritative record.
    //    If two requests race here, the @unique constraint fires on the second one (P2002).
    const booking = await prisma.booking.create({
      data: {
        name,
        email,
        phone: phone || null,
        message: message || null,
        slotTime: slotDate,
      },
    });

    // 5. Create a Google Calendar event — non-fatal.
    //    If the OAuth token hasn't been re-authorized for the new scope, this returns 403.
    //    The booking is still valid; the trainer can add the event manually.
    let calendarEventId: string | null = null;
    try {
      await createCalendarEvent({ name, email, slotTime: booking.slotTime });
      // Note: events.insert returns the event id — store it if you want to support cancellation later
    } catch {
      // intentionally non-fatal — log in production monitoring
    }

    // 6. Send visitor confirmation email — non-fatal.
    try {
      await sendVisitorConfirmation({ name, email, phone, message, slotTime: booking.slotTime });
    } catch {
      // intentionally non-fatal
    }

    // 7. Send trainer notification email — non-fatal.
    try {
      await sendTrainerNotification({ name, email, phone, message, slotTime: booking.slotTime });
    } catch {
      // intentionally non-fatal
    }

    res.status(201).json({ success: true, data: booking });
  } catch (err: unknown) {
    // P2002 = Prisma unique constraint violation on slotTime
    // This fires when two concurrent requests both pass the re-verify check above
    // and then race to the DB insert — the second one loses cleanly.
    if (err instanceof Prisma.PrismaClientKnownRequestError && err.code === 'P2002') {
      res.status(409).json({ success: false, error: 'This time slot was just booked. Please choose another.' });
      return;
    }
    res.status(500).json({ success: false, error: 'Failed to save booking.' });
  }
}
```

```typescript
import 'dotenv/config';
import express from 'express';
import cors from 'cors';
import session from 'express-session';
import rateLimit from 'express-rate-limit';
import contactRouter from './routes/contact.js';
import authRouter from './routes/auth.js';
import slotsRouter from './routes/slots.js';
import bookingsRouter from './routes/bookings.js';

const app = express();
const PORT = process.env.PORT || 3001;
const ALLOWED_ORIGIN = process.env.ALLOWED_ORIGIN || 'http://localhost:5173';

app.use(cors({ origin: ALLOWED_ORIGIN }));
app.use(express.json());

// Session middleware — used only to store the OAuth state param during the
// auth redirect flow. The state is a short-lived CSRF token; nothing sensitive
// is kept in the session long-term.
app.use(session({
  secret: process.env.SESSION_SECRET!,
  resave: false,
  saveUninitialized: false,
  cookie: {
    httpOnly: true,   // JS can't read the cookie — prevents XSS theft
    secure: process.env.NODE_ENV === 'production', // HTTPS-only in prod
    maxAge: 10 * 60 * 1000, // 10 minutes — enough to complete the OAuth flow
  },
}));

// General API rate limiter — applied to all /api/* routes (SEC-04).
// 100 requests per 15 minutes per IP is generous for normal use but stops scrapers.
const apiLimiter = rateLimit({
  windowMs: 15 * 60 * 1000, // 15 minutes
  max: 100,
  standardHeaders: true,  // sends RateLimit-* headers (RFC 6585)
  legacyHeaders: false,   // disables the old X-RateLimit-* headers
  message: { success: false, error: 'Too many requests. Please try again later.' },
});

// Stricter limiter for the booking endpoint — prevents form spam.
// 5 booking attempts per IP per hour is more than enough for legitimate users.
const bookingLimiter = rateLimit({
  windowMs: 60 * 60 * 1000, // 1 hour
  max: 5,
  standardHeaders: true,
  legacyHeaders: false,
  message: { success: false, error: 'Too many booking attempts. Please try again later.' },
});

// General limiter covers all /api routes — must be registered before route mounts
app.use('/api', apiLimiter);

// Stricter booking limiter layered on top of the general one
app.use('/api/bookings', bookingLimiter);

app.use('/api/contact', contactRouter);
app.use('/auth', authRouter);
app.use('/api/slots', slotsRouter);
app.use('/api/bookings', bookingsRouter);

app.listen(PORT);
```

Note: `app.use('/api/bookings', bookingLimiter)` must appear BEFORE `app.use('/api/bookings', bookingsRouter)` so the limiter runs first. Express processes middleware in registration order.
